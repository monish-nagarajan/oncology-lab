import streamlit as st
import matplotlib.pyplot as plt
import numpy as np

st.set_page_config(page_title="Cancer Cell Growth Lab", layout="centered")

st.title("🧬 Cancer Cell Growth Virtual Lab")

st.markdown("Simulate how normal and cancer cells grow under different conditions like nutrients and drug dosage.")

# Sidebar controls (professional touch 😏)
st.sidebar.header("🧪 Experiment Controls")

days = st.sidebar.slider("Time (days)", 1, 30, 10)
nutrients = st.sidebar.slider("Nutrient Level", 1, 10, 5)
drug = st.sidebar.slider("Drug Dose", 0, 10, 2)

# Time array
t = np.linspace(0, days, 100)

# Growth models
normal_growth = nutrients * (1 + 0.2 * t) - (drug * 0.3)
cancer_growth = nutrients * np.exp(0.15 * t) - (drug * 0.5)

# Avoid negative values
normal_growth = np.maximum(normal_growth, 0)
cancer_growth = np.maximum(cancer_growth, 0)

# Plot
fig, ax = plt.subplots()

ax.plot(t, normal_growth, label="Normal Cells")
ax.plot(t, cancer_growth, label="Cancer Cells")

ax.set_xlabel("Time (days)")
ax.set_ylabel("Cell Population")
ax.set_title("Cell Growth Simulation")
ax.legend()

st.pyplot(fig)

# Analysis Section
st.subheader("🧠 Observation")

if drug == 0:
    st.warning("⚠️ No drug applied — cancer cells grow uncontrollably.")
elif drug < 5:
    st.info("💊 Drug slows cancer growth slightly.")
else:
    st.success("💊 High drug dose effectively suppresses cancer growth.")

# Extra insight
if nutrients > 7:
    st.write("🍽️ High nutrients support faster cell growth.")
elif nutrients < 3:
    st.write("🥀 Low nutrients limit cell growth.")

# Footer
st.markdown("---")
st.caption("Virtual Biotechnology Lab | Cancer Cell Simulation")
